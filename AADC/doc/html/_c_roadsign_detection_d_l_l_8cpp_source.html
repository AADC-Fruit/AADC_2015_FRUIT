47 // helper function:
48 // finds a cosine of angle between vectors
49 // from pt0->pt1 and from pt0->pt2
50 static tFloat64 angle( Point pt1, Point pt2, Point pt0 )
51 {
52  tFloat64 dx1 = pt1.x - pt0.x;
53  tFloat64 dy1 = pt1.y - pt0.y;
54  tFloat64 dx2 = pt2.x - pt0.x;
55  tFloat64 dy2 = pt2.y - pt0.y;
56  return (dx1*dx2 + dy1*dy2)/sqrt((dx1*dx1 + dy1*dy1)*(dx2*dx2 + dy2*dy2) + 1e-10);
57 }
