181 tFloat32 ControllerFilter::getControllerValue(tFloat32 measuredValue)
182 {
183  //calculate the sample time in milliseceonds if necessary
184  tInt sampleTime;
185  if (GetPropertyBool("use automatically calculated sample interval",1)==tTrue)
186  sampleTime = tInt(GetTime() - m_lastSampleTime)*1000;
187  else
188  sampleTime = GetPropertyInt("Sample Intervall [msec]",1);
189  m_lastSampleTime = GetTime();
190 
191  //the three controller algorithms
192  if (GetPropertyInt("Controller Typ", 1)==1)
193  {
194  //P-Regler y = Kp * e
195  tFloat32 result = tFloat32(GetPropertyFloat("Controller Kp value",1)*(m_setPoint-measuredValue));
196  return result;
197  }
198  else if(GetPropertyInt("Controller Typ", 1)==2) //PI- Regler
199  {
200  //esum = esum + e
201  //y = Kp * e + Ki * Ta * esum
202  m_accumulatedVariable +=(m_setPoint-measuredValue);
203  tFloat32 result = tFloat32(GetPropertyFloat("Controller Kp value",1)*(m_setPoint-measuredValue) \
204  +GetPropertyFloat("Controller Ki value",1)*sampleTime*m_accumulatedVariable);
205  return result;
206  }
207  else if(GetPropertyInt("Controller Typ", 1)==3)
208  {
209  //esum = esum + e
210  //y = Kp * e + Ki * Ta * esum + Kd * (e  ealt)/Ta
211  //ealt = e
212  m_accumulatedVariable +=(m_setPoint-measuredValue);
213  tFloat32 result = tFloat32(GetPropertyFloat("Controller Kp value",1)*(m_setPoint-measuredValue) \
214  +tFloat32(GetPropertyFloat("Controller Ki value",1))*sampleTime*m_accumulatedVariable) \
215  +tFloat32(GetPropertyFloat("Controller Kd value",1))*((m_setPoint-measuredValue)-m_lastMeasuredError)/sampleTime;
216  m_lastMeasuredError = (m_setPoint-measuredValue);
217  return result;
218  }
219  else
220  return 0;
221 }
