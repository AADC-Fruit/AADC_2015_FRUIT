134 tResult cJuryTransmitter::OnPinEvent(IPin* pSource, tInt nEventCode, tInt nParam1, tInt nParam2, IMediaSample* pMediaSample)
135 {
136  RETURN_IF_POINTER_NULL(pMediaSample);
137  RETURN_IF_POINTER_NULL(pSource);
138 
139  if (nEventCode == IPinEventSink::PE_MediaSampleReceived && m_pCoderDescNotAus != NULL && pSource == &m_DriverStateInputPin)
140  {
141  // read-out the incoming Media Sample
142  cObjectPtr<IMediaCoder> pCoder;
143  RETURN_IF_FAILED(m_pCoderDescDriverState->Lock(pMediaSample, &pCoder));
144 
145  tBool stateDriver = false;
146  tInt16 entry = -1;
147 
148  pCoder->Get("bDriverState", (tVoid*)&stateDriver);
149  pCoder->Get("i16ManeuverEntry", (tVoid*)&entry);
150  m_pCoderDescDriverState->Unlock(pCoder);
151  emit sendDriverState(stateDriver, entry);
152  if ( m_last_timestamp !=0)
153  {
154  tTimeStamp timeDiff = (pMediaSample->GetTime() - m_last_timestamp)/1000; //time in milliseconds
155  m_last_timestamp = pMediaSample->GetTime();
156  QString Message;
157  if (stateDriver)
158  Message = "from " + QString::number(m_last_entryId) + " to " + QString::number(entry) + ": " + QString::number(timeDiff) + " msec: OK";
159  else
160  Message = "from " + QString::number(m_last_entryId) + " to " + QString::number(entry) + ": " + QString::number(timeDiff) + " msec: Error";
161  m_last_entryId = entry;
162  sendMessage(Message);
163  }
164  else
165  m_last_timestamp = pMediaSample->GetTime();
166 
167  if(m_bDebugModeEnabled) LOG_INFO(cString::Format("Jury Module: State value: %d with index %d",stateDriver,entry));
168 
169  }
170  RETURN_NOERROR;
171 
172 }
