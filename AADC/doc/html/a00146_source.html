278 tResult cArduinoAktors::ProcessActuatorValue( IMediaSample* pMediaSample, tUInt8 chID )
279 {
280  __synchronized_obj(m_oActuatorCritSection);
281  if (pMediaSample != NULL && m_pCoderDescSignalInput != NULL)
282  {
283  //write values with zero
284  tFloat32 f32Value = 0;
285  { // focus for sample read lock
286  // read-out the incoming Media Sample
287  __adtf_sample_read_lock_mediadescription(m_pCoderDescSignalInput,pMediaSample,pCoder);
288  //get values from media sample
289  pCoder->Get("f32Value", (tVoid*)&f32Value);
290  }
291 
292  // because the ARDUINO protocol defines the range between 0...180
293  // the values must be limited
294  if (f32Value > 180.0f)
295  {
296  f32Value = 180.0f;
297  LOG_WARNING("Actuator value out of range. Setting to max. 180");
298  }
299  if (f32Value < 0.0f)
300  {
301  f32Value = 0.0f;
302  LOG_WARNING("Actuator value out of range. Setting to min. 0");
303  }
304  // cast from float to tUInt8 with correct rounding
305  tUInt8 ui8Data = static_cast<tUInt8> (f32Value + 0.5f);
306  BuildArduinoFrame(chID,1,&ui8Data);
307  }
308  RETURN_NOERROR;

341 tResult cArduinoAktors::BuildArduinoFrame( tUInt8 ch8Id, tUInt8 i8DataLength, tUInt8 *pchData )
342 {
343  __synchronized_obj(m_oTransmitCritSection);
344  // build empty frame
345  tUInt8 achFrameData[25];
346  cMemoryBlock::MemSet(achFrameData, 0x00, sizeof(achFrameData));
347  cMemoryBlock::MemCopy(achFrameData, pchData, i8DataLength);
348 
349  //create new media sample
350  cObjectPtr<IMediaSample> pMediaSample;
351  RETURN_IF_FAILED(AllocMediaSample((tVoid**)&pMediaSample));
352 
353  //allocate memory with the size given by the descriptor
354  cObjectPtr<IMediaSerializer> pSerializer;
355  m_pCoderDescArduinoData->GetMediaSampleSerializer(&pSerializer);
356  tInt nSize = pSerializer->GetDeserializedSize();
357  RETURN_IF_FAILED(pMediaSample->AllocBuffer(nSize));
358 
359  { // focus for sample write lock
360  //write date to the media sample with the coder of the descriptor
361  __adtf_sample_write_lock_mediadescription(m_pCoderDescArduinoData,pMediaSample,pCoder);
362 
363  tUInt8 i8SOF = ID_ARD_SOF;
364  tUInt32 ui32ArduinoTimestamp = 0;
365 
366  pCoder->Set("ui8SOF", (tVoid*)&i8SOF);
367  pCoder->Set("ui8ID", (tVoid*)&ch8Id);
368  pCoder->Set("ui32ArduinoTimestamp", (tVoid*)&(ui32ArduinoTimestamp));
369 
370  pCoder->Set("ui8DataLength", (tVoid*)&i8DataLength);
371  pCoder->Set("ui8Data", (tVoid*)&(achFrameData));
372  }
373 
374  //transmit media sample over output pin
375  pMediaSample->SetTime(_clock->GetStreamTime());
376  RETURN_IF_FAILED(m_oArduinoCOMOutPin.Transmit(pMediaSample));
377  // check if output is enabled and id is not watchdog OR id is watchdog and watchdog output is enabled and output is enabled
378  if (i8DataLength > 0)
379  {
380  if ((m_bDebugModeEnabled && ch8Id != ID_ARD_ACT_WD_TOGGLE) || (ch8Id == ID_ARD_ACT_WD_TOGGLE && !(m_bDebugModeEnabledWatchdog) && m_bDebugModeEnabled) ) LOG_INFO(cString::Format("Actuatorfilter sent: ID %x DataLength %d First Value %d",ch8Id,i8DataLength,achFrameData[0]));
381  }
382  RETURN_NOERROR;
383 }
