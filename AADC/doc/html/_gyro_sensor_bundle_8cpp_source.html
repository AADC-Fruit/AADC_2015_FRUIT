78  tInt16 q_z = 0;

85  pCoder->Get("i16Q_z", (tVoid*)&q_z);

87  m_pCoderDescInput->Unlock(pCoder);
88 
89  //create and transmit output mediasamples and normalize them to 16384.0f (maximum value from raw value from sensor)

93  tFloat32 flQ_z =tFloat32(q_z)/16384.0f;
