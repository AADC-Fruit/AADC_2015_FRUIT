183 tResult cMarkerDetectFilter::ProcessVideo(adtf::IMediaSample* pISample)
184 {
185 
186  RETURN_IF_POINTER_NULL(pISample);
187 
188  //creating new media sample for output
189  cObjectPtr<IMediaSample> pNewSample;
190  RETURN_IF_FAILED(_runtime->CreateInstance(OID_ADTF_MEDIA_SAMPLE, IID_ADTF_MEDIA_SAMPLE, (tVoid**) &pNewSample));
191  RETURN_IF_FAILED(pNewSample->AllocBuffer(m_sOutputFormat.nSize));
192 
193  //creating new pointer for input data
194  const tVoid* l_pSrcBuffer;
195  Mat TheInputImage;
196  //receiving data from inputsample, and saving to inputFrame
197  if (IS_OK(pISample->Lock(&l_pSrcBuffer)))
198  {
199  //convert to mat
200  TheInputImage = Mat(m_sInputFormat.nHeight,m_sInputFormat.nWidth,CV_8UC3,(tVoid*)l_pSrcBuffer,m_sInputFormat.nBytesPerLine);
201  m_MDetector.detect(TheInputImage,m_TheMarkers,m_TheCameraParameters,m_TheMarkerSize);
202  }
203  pISample->Unlock(l_pSrcBuffer);
204  //print marker info and draw the markers in image
205  for (unsigned int i=0;i<m_TheMarkers.size();i++)
206  {
207  if (m_outputMode!=1) m_TheMarkers[i].draw(TheInputImage,Scalar(0,0,255),1);
208  markerDetected(m_TheMarkers[i].id, m_TheMarkers[i].getArea(),pISample->GetTime());
209  }
210 
211  //update new media sample with image data if something has ti be transmitted
212  if (m_outputMode!=1)
213  {
214  pNewSample->Update(pISample->GetTime(), TheInputImage.data, m_sOutputFormat.nSize, 0);
215  m_oPinOutputVideo.Transmit(pNewSample);
216  }
217  // send a sample which indicates that no sign was detectec in this frame
218  //if (!(m_SignInFrame)) sendRoadSignStruct(0,0,pISample->GetTime());

220  RETURN_NOERROR;
