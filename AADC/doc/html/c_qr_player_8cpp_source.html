131 tResult cQrPlayer::ProcessVideo(adtf::IMediaSample* pISample)
132 {
133  RETURN_IF_POINTER_NULL(pISample);
134 // creating new temporary variables
135  adtf_util::cImage oImage;
136  const tVoid* l_pSrcBuffer;
137 
138 // receiving data from inputsample, and saving to oImage
139  if (IS_OK(pISample->Lock(&l_pSrcBuffer)))
140  {
141  oImage.SetBits((tUInt8*) l_pSrcBuffer, &m_sInputFormat);
142 
143  pISample->Unlock(l_pSrcBuffer);
144  }
145 
146 // convert adtf-image to greyscaled image which is used by zbar
147  RETURN_IF_FAILED(oImage.Convert(&m_sProcessFormat));
148 
149 // creates image which is used by the QR Scanner
150  Image image(m_sInputFormat.nWidth, m_sInputFormat.nHeight, "Y800", (unsigned char*)oImage.GetBitmap(), m_sInputFormat.nWidth * m_sInputFormat.nHeight);
151 
152 // scan image for codes and check for errors
153  int n = m_qrScanner.scan(image);
154  if (n== -1) LOG_ERROR(cString::Format("QR Player: Error in QRScanner"));
155 
156 // extract results to console
157  for(Image::SymbolIterator symbol = image.symbol_begin();
158  symbol != image.symbol_end();
159  ++symbol) {
160  //print only new results
161  if (m_lastQRCode!=symbol->get_data())
162  {
163  m_lastQRCode=symbol->get_data();
164  }
165  ProcessCommand(cString(symbol->get_data().c_str()));
166  }
167 
168  // clean up
169  image.set_data(NULL, 0);
170 
171  RETURN_NOERROR;
172 }

239 
240  //create new media sample
241  cObjectPtr<IMediaSample> pMediaSample;
242  AllocMediaSample((tVoid**)&pMediaSample);
243 
244  //allocate memory with the size given by the descriptor

246  m_pCoderDescSignalOutput->GetMediaSampleSerializer(&pSerializer);
247  tInt nSize = pSerializer->GetDeserializedSize();
248  pMediaSample->AllocBuffer(nSize);
249

252  m_pCoderDescSignalOutput->WriteLock(pMediaSample, &pCoder);

256  m_pCoderDescSignalOutput->Unlock(pCoder);
257 
258  //transmit media sample over output pin
259  pMediaSample->SetTime(_clock->GetStreamTime());
