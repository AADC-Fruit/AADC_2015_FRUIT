34 void cRotationalOffset::CartisanToPolar( const Contour &contour )
35 {
36  //unsigned int i;
37 
38  int vXc = 0;
39  int vYc = 0;
40  int vX = 0;
41  int vY = 0;
42  float vRad = 0.0;
43  float vTheta = 0.0;
44  //mNoOfPoints = contour->nop;
45  //vector< pair<float, float> > CPtr = contour;
46 
47  Contour::const_iterator i;
48  //Calculate the Centre of Mass(COM)of the contour.
49 
50  for ( i = contour.begin(); i != contour.end(); i++ )
51  {
52  vXc += i->first;
53  vYc += i->second;
54  //contour = contour->next;
55  //contour++;//change here...above one is correct
56  }
57 
58  //Normalise the COM
59  vXc /= contour.size();//mNoOfPoints;
60 
61  vYc /= contour.size();//mNoOfPoints;
62 
63  //Restore the base pointer.
64  //contour = CPtr;
65 
66  //Translation of the origin to COM and convert from Cartisian to Polar co-ordinates
67  for ( i = contour.begin(); i != contour.end(); i++ )
68  {
69  vX = i->first - vXc;
70  vY = i->second - vYc;
71  vRad = static_cast<float>( sqrt( static_cast<double>( vX * vX ) +
72  static_cast<double>( vY * vY ) ) );
73 
74  vTheta = static_cast<float>( atan2( static_cast<double>( vY ),
75  static_cast<double>( vX ) ) ); /*Gives the output in range -PI to PI */
76 
77  //if theta lies in 3rd or 4th quadrant then make it in range of 0 - 2 * pi, add 2*PI to all theta values.
78 
79  if ( vY < 0 )
80  vTheta += 2 * static_cast<float>( cPI );
81 
82  /*Store information<theta,radius> in vector pair*/
83  mThetaRad.push_back( make_pair( vTheta, vRad ) );
84 
85  //contour = contour->next;
86  //contour++;//change here...above one is correct
87  }
88 }
