441 tResult cArduinoSensors::Process_ArduinoCOMInputPin(IMediaSample *pMediaSample)
442 {
443  if (pMediaSample != NULL && m_pCoderArduinoData != NULL)
444  {
445  // read-out the incoming Media Sample
446  cObjectPtr<IMediaCoder> pCoder;
447  RETURN_IF_FAILED(m_pCoderArduinoData->Lock(pMediaSample, &pCoder));
448 
449  //write values with zero
450  tInt8 SOF = 0;
451  tChar ID = 0;
452  tUInt32 arduinoTimestamp = 0;
453  tInt8 DataLength = 0;
454  tChar frameData[25];
455 
456  //get values from media sample
457  pCoder->Get("i8SOF", (tVoid*)&SOF);
458  pCoder->Get("chID", (tVoid*)&ID);
459  pCoder->Get("ui32ArduinoTimestamp", (tVoid*)&arduinoTimestamp);
460  pCoder->Get("i8DataLength", (tVoid*)&DataLength);
461  pCoder->Get("chData", (tVoid*)&(frameData));
462  if (m_bDebugModeEnabled) LOG_INFO(cString::Format("Sensorfilter received: ID %x DataLength %d",ID,DataLength));
463 
464  m_pCoderArduinoData->Unlock(pCoder);
465 
466  switch(ID)
467  {
468  case (ID_ARD_SENS_WHEELENC):
469  RETURN_IF_FAILED(TransmitCountRPMData(pMediaSample->GetTime(),arduinoTimestamp,frameData));
470  break;
471  case (ID_ARD_SENS_US):
472  RETURN_IF_FAILED(TransmitUsData(pMediaSample->GetTime(),arduinoTimestamp,frameData));
473  break;
474  case (ID_ARD_SENS_IMU):
475  RETURN_IF_FAILED(TransmitAccData(pMediaSample->GetTime(),arduinoTimestamp,frameData));
476  RETURN_IF_FAILED(TransmitGyroData(pMediaSample->GetTime(),arduinoTimestamp,frameData));
477  break;
478  case (ID_ARD_SENS_VOLTAGE):
479  RETURN_IF_FAILED(TransmitVoltData(pMediaSample->GetTime(),arduinoTimestamp,frameData));
480  break;
481  case (ID_ARD_SENS_PHOTO):
482  RETURN_IF_FAILED(TransmitPhotoData(pMediaSample->GetTime(),arduinoTimestamp,frameData));
