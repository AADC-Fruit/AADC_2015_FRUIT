73  115200, cSerialDevice::SER_NOPARITY, 8, cSerialDevice::SER_ONESTOPBIT,

160  //buffer
161  tUInt8 header[7];
162  tUInt8 payload[32];

164  cMemoryBlock::MemSet(payload, 0x00, sizeof(payload));
165 
166  tInt read = 0;
167  tUInt32 bytesAvail = m_oSerialDevice.BytesAvailable();
168  // length of largest data package
169  if( bytesAvail > 40)
170  {
171  m_timeLastDataReceived = GetStreamTime();
172  //look for a SOF flag indicating an header
173  do
174  {
175  read = m_oSerialDevice.Read(&header[0], 1);
176  if(read != 1) LOG_ERROR("Serial Communication: received data does not match read request");
177  } while(header[0] != ID_ARD_SOF);
178 
179  //read the header with 6 bytes
180  read = m_oSerialDevice.Read(header + 1, 6);
181  if(read != 6) LOG_ERROR("Serial Communication: received data does not match read request");
182 
183  //check whether length of data is valid
184  if(header[6] <= 32)
185  {
186  //read the payload
187  m_oSerialDevice.Read(payload, header[6]);
188
