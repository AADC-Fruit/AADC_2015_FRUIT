274 tResult cArduinoAktors::ProcessActuatorValue( IMediaSample* pMediaSample, tChar chID )
275 {
276  if (pMediaSample != NULL && m_pCoderDescSignalInput != NULL)
277  {
278  // read-out the incoming Media Sample
279  cObjectPtr<IMediaCoder> pCoder;
280  RETURN_IF_FAILED(m_pCoderDescSignalInput->Lock(pMediaSample, &pCoder));
281 
282  //write values with zero
283  tFloat32 f32Value = 0;
284 
285  //get values from media sample
286  pCoder->Get("f32Value", (tVoid*)&f32Value);
287 
288  m_pCoderDescSignalInput->Unlock(pCoder);
289 
290  // because the ARDUINO protocol defines the range between 0...180
291  // the values must be limited
292  if (f32Value > 180.0f)
293  {
294  f32Value = 180.0f;
295  LOG_WARNING("Actuator value out of range. Setting to max. 180");
296  }
297  if (f32Value < 0.0f)
298  {
299  f32Value = 0.0f;
300  LOG_WARNING("Actuator value out of range. Setting to min. 0");
301  }
302  // cast from float to tChar with correct rounding
303  tChar chData = static_cast<tChar> (f32Value + 0.5f);
304  BuildArduinoFrame(chID,1,&chData);
305  }
306  RETURN_NOERROR;

340 tResult cArduinoAktors::BuildArduinoFrame( tChar ch8Id, tInt8 i8DataLength, tChar *pchData )
341 {
342  // build empty frame
343  tChar achFrameData[25];
344  cMemoryBlock::MemSet(achFrameData, 0x00, sizeof(achFrameData));
345  cMemoryBlock::MemCopy(achFrameData, pchData, i8DataLength);
346 
347  //create new media sample
348  cObjectPtr<IMediaSample> pMediaSample;
349  RETURN_IF_FAILED(AllocMediaSample((tVoid**)&pMediaSample));
350 
351  //allocate memory with the size given by the descriptor
352  cObjectPtr<IMediaSerializer> pSerializer;
353  m_pCoderDescArduinoData->GetMediaSampleSerializer(&pSerializer);
354  tInt nSize = pSerializer->GetDeserializedSize();
355  RETURN_IF_FAILED(pMediaSample->AllocBuffer(nSize));
356 
357  //write date to the media sample with the coder of the descriptor
358  cObjectPtr<IMediaCoder> pCoder;
359  RETURN_IF_FAILED(m_pCoderDescArduinoData->WriteLock(pMediaSample, &pCoder));
360 
361  tInt8 i8SOF = ID_ARD_SOF;
362  tUInt32 ui32ArduinoTimestamp = 0;
363 
364  pCoder->Set("i8SOF", (tVoid*)&i8SOF);
365  pCoder->Set("chID", (tVoid*)&ch8Id);
366  pCoder->Set("ui32ArduinoTimestamp", (tVoid*)&(ui32ArduinoTimestamp));
367 
368  pCoder->Set("i8DataLength", (tVoid*)&i8DataLength);
369  pCoder->Set("chData", (tVoid*)&(achFrameData));
370  m_pCoderDescArduinoData->Unlock(pCoder);
371 
372  //transmit media sample over output pin
373  pMediaSample->SetTime(_clock->GetStreamTime());
374  RETURN_IF_FAILED(m_oArduinoCOMOutPin.Transmit(pMediaSample));
375  // check if output is enabled and id is not watchdog OR id is watchdog and watchdog output is enabled and output is enabled
376  if (i8DataLength > 0)
377  {
378  if ((m_bDebugModeEnabled && ch8Id != ID_ARD_ACT_WD_TOGGLE) || (ch8Id == ID_ARD_ACT_WD_TOGGLE && !(m_bDebugModeEnabledWatchdog) && m_bDebugModeEnabled) ) LOG_INFO(cString::Format("Actuatorfilter sent: ID %x DataLength %d First Value %d",ch8Id,i8DataLength,achFrameData[0]));
379  }
380  RETURN_NOERROR;
381 }
