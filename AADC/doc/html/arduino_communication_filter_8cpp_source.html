66 tInt cArduinoCOM::ReceiveFrameFromArduino(tArduinoFrame *psFrame, const tInt8 i8MaxFrameLen)
67 {
68 
69  if(psFrame == 0 || i8MaxFrameLen == 0)
70  {
71  return 0;
72  }
73 
74 
75  tChar *pchFrame = (tChar*) (psFrame);
76 
77  tInt nFrameBytesToRead = 1;
78  tInt nFrameBytesRead = 0;
79 
80 
81 
82  //Search SOF
83  while(nFrameBytesRead < nFrameBytesToRead)
84  {
85  tInt nReadFromDev = m_oDevice.Read(& (pchFrame[nFrameBytesRead]),1);
86  if(nReadFromDev != 1)
87  {
88  tTimeStamp tsCurrentTime = _clock->GetStreamTime();
89 
90  //only one message every 2000 ms
91  if( (tsCurrentTime - m_tsDebugTimestamp) > 2000 * 1000)
92  {
93  LOG_INFO(cString::Format("Frame SOF mismatch"));
94  m_tsDebugTimestamp = tsCurrentTime;
95  }
96  }
97  if(psFrame->sHeader.i8SOF == ID_ARD_SOF)
98  {
99  // SOF found
100  nFrameBytesRead = 1;
101  }
102  }
103 
104  //read hole header
105  nFrameBytesToRead = 7;
106 
107 
108  while(nFrameBytesRead < nFrameBytesToRead)
109  {
110  tInt nReadFromDevice = m_oDevice.Read(& (pchFrame[nFrameBytesRead]),nFrameBytesToRead - nFrameBytesRead);
111  if(nReadFromDevice <= 0 || nReadFromDevice > (nFrameBytesToRead - nFrameBytesRead))
112  {
113  LOG_INFO(cString::Format("Frame Header mismatch %i",nReadFromDevice));
114  return 0;
115  }
116  else
117  {
118  nFrameBytesRead += nReadFromDevice;
119  }
120  }
121 
122  tInt nDataFrameLen = psFrame->sHeader.i8DataLength;
123  if(nDataFrameLen == 0 || nDataFrameLen > 25)
124  {
125  LOG_INFO(cString::Format("Frame DataLen mismatch %i",nDataFrameLen));
126  return 0;
127  }
128 
129  //read data
130 
131  nFrameBytesToRead += nDataFrameLen;
132 
133  while(nFrameBytesRead < nFrameBytesToRead)
134  {
135  tInt nReadFromDevice = m_oDevice.Read(& (pchFrame[nFrameBytesRead]),nFrameBytesToRead - nFrameBytesRead);
136  nFrameBytesRead += nReadFromDevice;
137  }
138 
139 
140  // LOG_INFO(cString::Format("Frame complete ID %x len %i",frame->header.id, frame->header.dataLen));
141 
142  return nFrameBytesRead;
143  }

358 tResult cArduinoCOM::ProcessInput(IMediaSample *pMediaSample)

361  {

365 
366  //write values with zero
367  tInt8 i8SOF = 0;

369  tUInt32 ui32ArduinoTimestamp = 0;

375  pCoder->Get("chID", (tVoid*)&chID);
376  pCoder->Get("ui32ArduinoTimestamp", (tVoid*)&ui32ArduinoTimestamp);
377  pCoder->Get("i8DataLength", (tVoid*)&i8DataLength);
378  pCoder->Get("chData", (tVoid*)&(achFrameData));
379 
380  if (m_bDebugModeEnabled) LOG_INFO(cString::Format("Communication Filter send to arduino: ID %x DataLength %d",chID,i8DataLength));
381 
382  m_pCoderDescArduinoInputData->Unlock(pCoder);
383 
384  // construct frame
385  tChar chBuffer[32];
386  cMemoryBlock::MemSet(chBuffer,0x00,32);
387  tArduinoFrame *psFrame = (tArduinoFrame *) chBuffer;
388 
389  psFrame->sHeader.i8SOF = i8SOF;
390  psFrame->sHeader.chID = chID;
391  psFrame->sHeader.ui32ArduinoTimestamp = 0;
392  psFrame->sHeader.i8DataLength = i8DataLength;
