467 void RationalSuperShape2D::Optimize(const vector<Vector2d> &data, double &err,
468  bool normalization, int functionUsed) {

470  unsigned int i,k;
471  bool stop(false);

473  0,0,0,0,0,
474  0,0,0,0,0,

478 
479  MatrixXd alpha, alpha2;
480  VectorXd beta, beta2, dJ, oldBeta, trial, sigma;
481

483  alpha2 = MatrixXd::Zero(5,5);
484 
485  beta = VectorXd::Zero(5);
486  beta2 = VectorXd::Zero(5);
487  dJ = VectorXd::Zero(5);
488  oldBeta = VectorXd::Zero(5);
489  trial = VectorXd::Zero(8);
490  sigma = VectorXd::Zero(8);
491 
492  // derivatives of translation matrix are
493  // -constant and can be computed once for all
494  for(int itNum = 0; itNum < 1000 && stop == false; itNum++) {
495  bool outofBounds(false);
496  // store oldparams
497  for(i = 0; i < mParameters.size(); i++)
498  oldParams[i] = mParameters[i];
499 
500  alpha.setZero();
501  beta.setZero();
502  alpha2.setZero();
503  beta2.setZero();
504 
505  chiSquare = XiSquare(data,
506  alpha,
507  beta,
508  itNum == 0, // init x0, y0, and tht0 or not?
509  false, // robust or not?
510  functionUsed, // implicitf cuntion1
511  normalization, // activate normalization or not
512  true); // update vectors
513 
514  // add Lambda to diagonal elements
515  // -and solve the matrix
516  // linearization of Hessian, cf Numerical Recepies
517  for(k = 0; k < 5; k++) {
518  // multiplicative factor to make diagonal dominant
519  alpha(k,k) *= 1.0 + lambda;
520  // additive factor to avoid rank deficient matrix
521  alpha(k,k) += lambda;
522  }
523 
524  // solve system
525  alpha.ldlt().solveInPlace(beta);
526  // coefficients a and b in [0.01, 100]
527  outofBounds = mParameters[0] + beta[0] < 0.01 || mParameters[0] + beta[0] > 1000 ||
528  mParameters[1] + beta[1] < 0.01 || mParameters[1] + beta[1] > 1000 ||
529  mParameters[2] + beta[2] < 0.1 || mParameters[2] + beta[2] > 1000 ||
530  mParameters[3] + beta[3] < 0.001 || mParameters[3] + beta[3] > 1000 ||
531  mParameters[4] + beta[4] < 0.001 || mParameters[4] + beta[4] > 1000;
532 
533  if( !outofBounds )
534  {
535  Set_a(mParameters[0] + beta[0]);
536  Set_b(mParameters[1] + beta[1]);
537  // coefficients n1 in [1., 1000]
538  // setting n1<1. leads to strong
539  // -numerical instabilities
540  Set_n1( mParameters[2] + beta[2] );
541  // coefficients n2,n3 in [0.001, 1000]
542  Set_n2( mParameters[3] + beta[3]);
543  Set_n3( mParameters[4] + beta[4]);
544  // coefficients x0 and y0
545  // truncate translation to avoid huge gaps
546  // beta[5] = min(0.01, max(-0.01, beta[5]));
547  // beta[6] = min(0.01, max(-0.01, beta[6]));
548  // beta[7] = min(RATIONAL_PI/100.0, max(-RATIONAL_PI/100.0, beta[7]));
549  // mParameters[9] += beta[5];
550  // mParameters[10] += beta[6];
551  // rotational offset tht0
552  // mParameters[7] += beta[7];
553  }
554 
555  // evaluate chisquare with new values
556  oldChiSquare = chiSquare;
557  newChiSquare = XiSquare(data,
558  alpha2,
559  beta2,
560  false, // init x0, y0, and tht0 or not?
561  false, // robust or not?
562  functionUsed, // implicitf cuntion1
563  normalization, // actiate normalization or not
564  false);
565 
566  // check if better result
567  // if new result is not right
568  // -restore old params and try
569  // -with lambda 10 times bigger
570  if( newChiSquare > (0.999 * oldChiSquare) ) {
571  lambda *= lambdaIncr;
572  for(i = 0; i < mParameters.size(); i++) {
573  mParameters[i] = oldParams[i];
574  }
575  }
576  else {
577  // successful iteration
578 
579  // huge improvement, something may have been wrong
580  // this may arise during the first iterations
581  // n1 may literally explode, or tend to 0...
582  // in such case, the next iteration is successful but leads to a local minimum
583  // ==> it is better to verify the result with a smaller step
584  // if indeed it was a correct iteration, then it will pass the next time
585 
586  // 99% improvement, impossible
587  if (newChiSquare <= 0.01 * oldChiSquare) {
588  // reduce the step within the search direction
589  lambda *= lambdaIncr;
590  for(i = 0; i < mParameters.size(); i++) {
591  // restore old parameters
592  mParameters[i] = oldParams[i];
593  }
594  }
595  else {
596  // correct and realistic improvement
597  lambda /= lambdaIncr;
598  }
599  }
600  // very small displacement ==> local convergence
601  if( lambda > 1e15 || newChiSquare < 1e-5) {
602  stop = true;
603  }
604  } //end of for
605  err = chiSquare;
606 }

696 double RationalSuperShape2D::XiSquare(const vector < Vector2d > &data, MatrixXd &alpha,
697  VectorXd &beta, bool initOn,
698  bool robust, int functionUsed,
699  bool normalization, bool update) {
700  VectorXd dj;
701  vector<double> dF;
702 
703  Matrix3d trMatrix; // translational matrix
704  Matrix3d rotMatrix; // rotational matrix
705  Matrix3d dTrDx0; // partial derivative of the translational matrix regarding x
706  Matrix3d dTrDy0; // partial derivative of the translational matrix regarding y
707  Matrix3d dRotDtht0; // partial derivative of rotational matrix regarding theta
708 
709  unsigned int i,j,k;
710 
711  double r, x, y, tht, dThtDx, dThtDy,dThtDx0, dThtDy0, dRdTht,
712  chiSquare(1e15),
713  f(0),h(0),
714  x0(Get_xoffset()),y0(Get_yoffset()),tht0(Get_thtoffset()),
715  dXdX0,dXdY0,dXdTht0, dYdX0,dYdY0,dYdTht0, dThtdTht0;
716 
717  // derivatives of translation matrix are
718  // -constant and can be computed once for all
719  dTrDx0 << 0, 0, -1,
720  0, 0, 0,
721  0, 0, 1;
722  dTrDy0 << 0, 0, 0,
723  0, 0, -1,
724  0, 0, 1;
725 
726  dj = VectorXd::Zero(8);
727 
728  // functions pointer
729  double (RationalSuperShape2D ::*pt2ConstMember)( const Vector2d &P, vector <double> &DfFinal ) = NULL;
730 
731  switch (functionUsed)
732  {
733  case 1 : {
734  pt2ConstMember = &RationalSuperShape2D :: ImplicitFunction1;
735  } break;
736 
737  case 2 :{
738  pt2ConstMember = &RationalSuperShape2D :: ImplicitFunction2;
739  } break;
740 
741  case 3 : {
742  pt2ConstMember = &RationalSuperShape2D :: ImplicitFunction3;
743  } break;
744 
745  default :
746  pt2ConstMember = &RationalSuperShape2D :: ImplicitFunction1;
747  }
748 
749  if(initOn)
750  {
751  // THIS IS ONLY CALLED AT THE VERY FIRST ITERATION
752  if(robust) {
753  RobustInit(data);
754  }
755  else
756  {
757  // Finding the center of gravity
758  Vector2d center(0,0);
759  for(i = 0; i < data.size(); i++){
760  center += data[i];
761  }
762 
763  center /= data.size();
764 
765  double aMin(1e30);
766  for(i = 0; i < data.size(); i++) {
767  aMin = min(aMin, (data[i]-center).norm());
768  }
769  Set_a(aMin);
770  Set_b(aMin);
771  Set_xoffset(center[0]);
772  Set_yoffset(center[1]);
773  }
774  }
775 
776  // clean memory
777  if(update) {
778  alpha.setZero();
779  beta.setZero();
780  }
781 
782  // evaluate ChiSquare, components
783  // -for the beta and matrix alpha
784  chiSquare = 0;
785 
786  // first define inverse translation T-1
787  trMatrix.setZero();
788  rotMatrix.setZero();
789  dRotDtht0.setZero();
790 
791  trMatrix << 1 , 0 , -x0 ,
792  0 , 1 , -y0 ,
793  0 , 0 , 1;
794 
795  // get partial derivatives
796  // then define inverse rotation,
797  // -i.e. transposed rotation
798  rotMatrix << cos(tht0) , sin(tht0) , 0 ,
799  -sin(tht0) , cos(tht0) , 0 ,
800  0 , 0 , 1;
801 
802  // get partial derivatives
803  dRotDtht0 << -sin(tht0) , cos(tht0) , 0 ,
804  -cos(tht0) , -sin(tht0) , 0 ,
805  0 , 0 , 1;
806 
807  for(i = 0; i < data.size(); i++)
808  {
809  double dfDr;
810 
811  // global inverse transform is T * R
812  Vector2d dum(data[i]);
813  Vector3d dum2(dum[0],dum[1],1);
814  // apply inverse transform
815  Vector3d dum3 ( rotMatrix * (trMatrix * dum2));
816  // get partial derivatives for canonical point
817  Vector3d dPdX0 ( rotMatrix * (dTrDx0 * dum2) );
818  Vector3d dPdY0 ( rotMatrix * (dTrDy0 * dum2) );
819  Vector3d dPdTht0 (dRotDtht0 * (trMatrix * dum2) );
820  // the canonical point
821  Vector2d P(dum3[0],dum3[1]);
822  // simplify notations
823  dXdX0 = dPdX0[0];
824  dXdY0 = dPdY0[0];
825  dXdTht0 = dPdTht0[0];
826 
827  dYdX0 = dPdX0[1];
828  dYdY0 = dPdY0[1];
829  dYdTht0 = dPdTht0[1];
830 
831  x = P[0];
832  y = P[1];
833 
834  // avoid division by 0 ==> numerical stability
835 
836  if (P.norm() < RATIONAL_EPSILON) {
837  // avoids division by zero
838  continue;
839  }
840 
841  tht = atan2(y,x);
842  if(tht < 0) {
843  tht += 2.0 * RATIONAL_PI;
844  }
845 
846  r = Radius(tht);
847 
848  // theta = Arctan(Y/X)
849  // -y / (x*x+y*y);
850  dThtDx = -sin(tht) ;
851  // x / (x*x+y*y);
852  dThtDy = cos(tht);
853 
854  // partial derivatives of theta regarding
855  // - x offset, y offset, and angular offset
856  dThtDx0 = dThtDx * dXdX0 + dThtDy * dYdX0;
857  dThtDy0 = dThtDx * dXdY0 + dThtDy * dYdY0;
858  dThtdTht0 = dThtDx * dXdTht0 + dThtDy * dYdTht0;
859 
860  // avoid non differentiable cases
861  dRdTht = DrDtheta(tht);
862  // call to the implicit function
863  f = (*this.*pt2ConstMember)(P, dF);
864 
865  // compute elements beta[i][0] and alpha[i][j]
866  // ==> requires partial derivatives!!
867  // dF/dR stored at index 2 in array Df
868  // -during the call to ImplicitFunction1-2-3
869  dfDr = dF[2];
870 
871  // F1 = R-PL ==> DfDr = 1. ;
872  // F2 = 1-PL/R ==> DfDr = PL/R ;
873  // F3 = log ( R/PSL) ==> DfDr = 2/R
874  dj.setZero();
875  // df/da = df/dr * dr/da
876  dj[0] = dfDr * DrDa(tht) ;
877  // df/db = df/dr * dr/db
878  dj[1] = dfDr * DrDb(tht) ;
879  // df/dn1 = df/dr * dr/dn1
880  dj[2] = dfDr * DrDn1(tht);
881  // df/dn3 = df/dr * dr/dn3
882  dj[4] = dfDr * DrDn3(tht);
883  // df/dn2 = df/dr * dr/dn2
884  dj[3] = dfDr * DrDn2(tht);
885  // df/dx0 = df/dr * dr/dtht *dtht/dx0
886  //dj[5]= dfDr * dRdTht*dThtDx0;
887  // df/dy0 = df/dr * dr/dtht *dtht/dy0
888  //dj[6]= dfDr * dRdTht*dThtDy0;
889  // df/dth0 = dfdr * dr/dtht * dtht/dtht0
890  //dj[7]= dfDr * dRdTht*dThtdTht0;
891 
892  // df/dx + df/dy
893  double nablamagn = dF[0]*dF[0]
894  + dF[1]*dF[1];
895 
896  h = f * pow(f*f + nablamagn,-0.5);
897 
898  if(normalization) {
899  chiSquare += h*h;
900  }
901  else {
902  chiSquare += f*f;
903  }
904 
905  // scale factor for the gradient
906  // -if normalizing the function
907  if(update) {
908  if(normalization) {
909  if (fabs(f) > RATIONAL_EPSILON || nablamagn > RATIONAL_EPSILON ) {
910  for (int idum = 0; idum < 5; idum++){
911  dj[idum] *= h * (1.0 - h * h) / f;
912  }
913  }
914  else {
915  dj.setZero();
916  }
917  }
918  // summation of values for the gradient
919  if(normalization) {
920  if (f != 0) {
921  for(k = 0; k < 5; k++) {
922  beta[k] -= h * dj[k];
923  }
924  }
925  }
926  else {
927  for(k = 0; k < 5; k++) {
928  beta[k] -= f * dj[k];
929  }
930  }
931  // compute approximation of Hessian matrix
932  for(k = 0; k < 5; k++) {
933  for(j = 0; j < 5; j++) {
934  // Hessian Calculation
935  alpha(k,j) += dj[k] * dj[j];
936  }
937  }
938  }
939  }
940 
941  return chiSquare;
942 }
