20 retrieve_theta(tUInt r, tUInt g, tUInt b)
21 {
22  tFloat32 theta;
23 
24  // The numerator part of equation
25  tFloat32 numerator = r - (g * 0.5f) - (b * 0.5f);
26 
27  // The denominator part of equation
28  tFloat32 denominator = static_cast<tFloat32> ((r * r) + (g * g) + (b * b) - (r * g) - (r * b) - (g * b));
29 
30  tFloat32 temp = numerator / sqrtf(denominator);
31  theta = acos(temp);
32 
33  return theta;
34 }
