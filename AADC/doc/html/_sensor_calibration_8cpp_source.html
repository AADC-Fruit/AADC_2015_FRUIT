250 tFloat32 SensorCalibration::getLinearInterpolatedValue(tFloat32 inputValue)
251 {
252  if (inputValue<m_xValues.front())
253  {
254  if (m_bDebugModeEnabled) LOG_WARNING(adtf_util::cString::Format("SensorCalibration: requested x-value %f is lower than smallest x-value in calibration table",inputValue));
255  return m_yValues.front();
256  }
257  else if (inputValue>m_xValues.back())
258  {
259  if (m_bDebugModeEnabled) LOG_WARNING(adtf_util::cString::Format("SensorCalibration: requested x-value %f is higher than highes x-value in calibration table",inputValue));
260  return m_yValues.back();
261  }
262 
263  tUInt index;
264  if (m_xValues.size() > 2)
265  {
266  for (index = 0; index<m_xValues.size();index++)
267  {
268  if (m_xValues[index]>=inputValue) break;
269  }
270  if (index!=0) index = index -1;
271  //index = std::distance(m_xValues.begin(), std::lower_bound(m_xValues.begin(), m_xValues.end(), inputValue));
272  }
273 
274  else index = 0;
275 
276  if ((m_xValues[index+1]-m_xValues[index])!=0)
277  {
278  tFloat32 value = (inputValue-m_xValues[index])*(m_yValues[index+1]-m_yValues[index])/(m_xValues[index+1]-m_xValues[index])+m_yValues[index];
279 
280  //tFloat64 security check
281  if (value > *max_element(m_yValues.begin(),m_yValues.end() ))
282  return *max_element(m_yValues.begin(),m_yValues.end()) ;
283  else if (value < *min_element(m_yValues.begin(),m_yValues.end()) )
284  return *min_element(m_yValues.begin(),m_yValues.end()) ;
285  else
286  return value;
287  }
288  else
289  {
290  LOG_ERROR("SensorCalibration: divison by zero in linear interpolation!");
291  return 0;
292  }
