696 tResult cObjectDetection::ConvertRealWorldToProjectiveOffline(tInt32 nCount, const tFloatVector3D* aRealWorld, tFloatVector3D* aProjective)
697 {
698  //refer to original function xnConvertRealWorldToProjective in XnOpenNI.cpp
699  tFloat64 fXToZ = 1.122133; //values found by debugging the OPENNI Source
700  tFloat64 fYToZ = 0.84160; //values found by debugging the OPENNI Source
701 
702  tFloat64 fCoeffX = m_sInputFormatVideo.nWidth / fXToZ;
703  tFloat64 fCoeffY = m_outputMode.nYRes / fYToZ;
704 
705  // we can assume resolution is even (so integer div is sufficient)
706  tInt32 nHalfXres =m_sInputFormatVideo.nWidth / 2;
707  tInt32 nHalfYres =m_sInputFormatVideo.nHeight / 2;
708 
709  for (tInt32 i = 0; i < nCount; ++i)
710  {
711  aProjective[i].X = (tFloat32)fCoeffX * aRealWorld[i].X / aRealWorld[i].Z + nHalfXres;
712  aProjective[i].Y = nHalfYres - (tFloat32)fCoeffY * aRealWorld[i].Y / aRealWorld[i].Z;
713  aProjective[i].Z = aRealWorld[i].Z;
714  }
715 
716  RETURN_NOERROR;
717 }
