58 Mat PostProcessing::FilterImage()
59 {
60  //Morphological operation
61  //dilate
62  Mat sd=getStructuringElement(MORPH_CROSS,Size(4,4)); //structuring element used for dilate and erode
63  dilate(image,image,sd);
64 
65  Mat dst=image.clone(); //destination image; copy of the original image
66  threshold(image,image,254,255,CV_THRESH_BINARY); // make the image binary
67 
68  // find the contours of the objects
69  vector<vector<Point> > contours;
70  vector<Vec4i> hierarchy;
71  findContours(image, contours,hierarchy,CV_RETR_EXTERNAL, CV_CHAIN_APPROX_SIMPLE);
72 
73  //draw the filled contours to the destination image
74  Scalar color(255,255,255);
75  drawContours(dst, contours, -1, color, CV_FILLED, 8);
76 
77  //Morphological operations
78  //dilate
79  erode(dst,dst,sd);
80 
81  //eliminate nois using a median filter
82  for (int i=0;i<5;i++)
83  medianBlur(dst,dst,5);
84 
85  //output
86  return dst;
87 }

317 float PostProcessing:: distance( Point po, Point pf, Point pc )
318 {
319  float pox = ( float )po.x;
320  float poy = ( float )po.y;
321  float pfx = ( float )pf.x;
322  float pfy = ( float )pf.y;
323  float pcx = ( float )pc.x;
324  float pcy = ( float )pc.y;
325 
326  // In this function, we will compute the altitude of the triangle form by the two points of the convex hull and the one of the contour.
327  // It will allow us to remove points far of the convex conserving a degree of freedom
328 
329  // Compute the three length of each side of the triangle
330  // a will be the base too
331  float a = sqrt(pow(pfx - pox, 2.00f) + pow(pfy - poy, 2.00f));
332  // Compute the two other sides
333  float b = sqrt(pow(pcx - pox, 2.00f) + pow(pcy - poy, 2.00f));
334  float c = sqrt(pow(pfx - pcx, 2.00f) + pow(pfy - pcy, 2.00f));
335 
336  // Compute S which is the perimeter of the triangle divided by 2
337  float s = (a + b + c) / 2.00;
338 
339  // Compute the area of the triangle
340  float area = sqrt( s*(s - a)*(s - b)*(s-c) );
341 
342  // Compute the altitude
343  float altitude = 2 * area / a;
344 
345  return altitude;
346 
347  //return abs(( pfx - pox ) * ( pcy - poy ) - ( pfy - poy ) * ( pcx - pox ) ) / sqrt( pow(( pfx - pox ), 2 ) + pow(( pfy - poy ), 2 ) );
348 }
