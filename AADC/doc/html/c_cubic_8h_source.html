29 Cubic(tInt n, vector<tFloat32> x, vector<tFloat32> y)
30 {
31  m_n = n;
32  m_x = new tFloat64[n + 1];
33  m_y = new tFloat64[n + 1];
34  m_b = new tFloat64[n + 1];
35  m_c = new tFloat64[n + 1];
36  m_d = new tFloat64[n + 1];
37 
38  // remember and shift base and shift base.
39  for (tInt i = 1; i <= n; ++i) {
40  m_x[i] = x[i - 1];
41  m_y[i] = y[i - 1];
42  }
43 
44  // linear interpolation when we have too little data (i.e. just two points!)
45  if (n < 3) {
46  m_b[2] = m_b[1] = (m_y[2] - m_y[1]) / (m_x[2] - m_x[1]);
47  m_c[1] = m_c[2] = m_d[1] = m_d[2] = 0.0;
48  return;
49  }
50 
51  m_d[1] = m_x[2] - m_x[1];
52  m_b[1] = - m_d[1];
53  m_c[2] = (m_y[2] - m_y[1]) / m_d[1];
54  m_c[1] = 0.0;
55  for (tInt i = 2; i < n; ++i) {
56  m_d[i] = m_x[i + 1] - m_x[i];
57  m_b[i] = 2.0 * (m_d[i - 1] + m_d[i]);
58  m_c[i + 1] = (m_y[i + 1] - m_y[i]) / m_d[i];
59  m_c[i] = m_c[i + 1] - m_c[i];
60  }
61 
62  m_b[n] = -m_d[n - 1];
63  m_c[n] = 0.0;
64  if (n != 3) {
65  m_c[1] = m_c[3] / (m_x[4] - m_x[2]) - m_c[2] / (m_x[3] - m_x[1]);
66  m_c[n] = m_c[n - 1] / (m_x[n] - m_x[n - 2]) - m_c[n - 2] / (m_x[n - 1] - m_x[n - 3]);
67  m_c[1] *= m_d[1] * m_d[1] / (m_x[4] - m_x[1]);
68  m_c[n] *= - m_d[n - 1] * m_d[n - 1] / (m_x[n] - m_x[n - 3]);
69  }
70  for (tInt i = 2; i < n; ++i) {
71  tFloat64 T = m_d[i - 1] / m_b[i - 1];
72  m_b[i] -= T * m_d[i - 1];
73  m_c[i] -= T * m_c[i - 1];
74  }
75 
76  m_c[n] /= m_b[n];
77  for (tInt i = n - 1; i > 0; --i)
78  m_c[i] = (m_c[i] - m_d[i] * m_c[i + 1]) / m_b[i];
79 
80  m_b[n] = (m_y[n] - m_y[n - 1]) / m_d[n - 1] + m_d[n - 1] * (m_c[n - 1] + 2.0 * m_c[n]);
81 
82  for (tInt i = 1; i < n; ++i) {
83  m_b[i] = (m_y[i + 1] - m_y[i]) / m_d[i] - m_d[i] * (m_c[i + 1] + 2.0 * m_c[i]);
84  m_d[i] = (m_c[i + 1] - m_c[i]) / m_d[i];
85  m_c[i] *= 3.0;
86  }
87  m_c[n] *= 3.0;
88  m_d[n] = m_d[n - 1];
89 }

103 tFloat64 getValue(tFloat64 x)
104 {
105  if (x <= m_x[1])
106  return (((m_y[2]-m_y[1])/(m_x[2]-m_x[1]))*(x-m_x[1])+m_y[1]);
107  if (x >= m_x[m_n])
108  return (((m_y[m_n]-m_y[m_n-1])/(m_x[m_n]-m_x[m_n-1]))*(x-m_x[m_n-1])+m_y[m_n-1]);
109 
110  if (x <= m_x[2])
111  {
112  tFloat64 dx = x - m_x[1];
113  return m_y[1] + dx * (m_b[1] + dx * (m_c[1] + dx * m_d[1]));
114  }
115 
116  tInt i = 1, j = m_n + 1;
117  do {
118  tInt k = (i + j) / 2;
119  (x < m_x[k]) ? j = k : i = k;
120  } while (j > i + 1);
121 
122  tFloat64 dx = x - m_x[i];
123  return m_y[i] + dx * (m_b[i] + dx * (m_c[i] + dx * m_d[i]));
124 }
