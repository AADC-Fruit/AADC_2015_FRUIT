158 tResult cStateController::OnPinEvent(IPin* pSource, tInt nEventCode, tInt nParam1, tInt nParam2, IMediaSample* pMediaSample)
159 {
160 
161  RETURN_IF_POINTER_NULL(pMediaSample);
162  RETURN_IF_POINTER_NULL(pSource);
163 
164  if (nEventCode == IPinEventSink::PE_MediaSampleReceived )
165  {
166  //process the request to set state error pin
167  if (pSource == & m_StateErrorInputPin)
168  {
169  tBool bValue = tFalse;
170  { // focus for sample read lock
171  __adtf_sample_read_lock_mediadescription(m_pCoderDescSetStateError,pMediaSample,pCoder);
172  pCoder->Get("bValue", (tVoid*)&bValue);
173  }
174  if (bValue) changeState(stateCar_ERROR);
175  }
176  //process the request to set state running pin
177  else if (pSource == & m_StateRunningInputPin)
178  {
179  tBool bValue = tFalse;
180  { // focus for sample read lock
181  __adtf_sample_read_lock_mediadescription(m_pCoderDescSetStateRunning,pMediaSample,pCoder);
182  pCoder->Get("bValue", (tVoid*)&bValue);
183  }
184  if (bValue) changeState(stateCar_RUNNING);
185  }
186  //process the request to set state stop pin
187  else if (pSource == & m_StateCompleteInputPin)
188  {
189  tBool bValue = tFalse;
190  { // focus for sample read lock
191  __adtf_sample_read_lock_mediadescription(m_pCoderDescSetStateComplete,pMediaSample,pCoder);
192  pCoder->Get("bValue", (tVoid*)&bValue);
193  }
194  if (bValue) changeState(stateCar_COMPLETE);
195  }
196  //process the request to set state ready pin
197  else if (pSource == & m_StateReadyInputPin)
198  {
199  tBool bValue = tFalse;
200  { // focus for sample read lock
201  __adtf_sample_read_lock_mediadescription(m_pCoderDescSetStateReady,pMediaSample,pCoder);
202  pCoder->Get("bValue", (tVoid*)&bValue);
203  }
204  if (bValue) changeState(stateCar_READY);
205  }
206  //process the increment of maneuver id pin
207  else if (pSource == &m_StateIncrementManeuverInputPin)
208  {
209  tBool bValue = tFalse;
210  { // focus for sample read lock
211  __adtf_sample_read_lock_mediadescription(m_pCoderDescSetStateReady,pMediaSample,pCoder);
212  pCoder->Get("bValue", (tVoid*)&bValue);
213  }
214  if (bValue) incrementManeuverID();
215  }
216 
217  //process the increment of restart section pin
218  else if (pSource == &m_StateRestartSectionInputPin)
219  {
220  tBool bValue = tFalse;
221  { // focus for sample read lock
222  __adtf_sample_read_lock_mediadescription(m_pCoderDescSetStateReady,pMediaSample,pCoder);
223  pCoder->Get("bValue", (tVoid*)&bValue);
224  }
225  if (bValue) resetSection();
226  }
227 
228 
229  //process the request to the jury struct input pin
230  else if (pSource == &m_JuryStructInputPin)
231  {
232  tInt8 i8ActionID = -2;
233  tInt16 i16entry = -1;
234 
235  { // focus for sample read lock
236  __adtf_sample_read_lock_mediadescription(m_pCoderDescJuryStruct,pMediaSample,pCoder);
237 
238  pCoder->Get("i8ActionID", (tVoid*)&i8ActionID);
239  pCoder->Get("i16ManeuverEntry", (tVoid*)&i16entry);
240  }
241  //change the state depending on the input
242  // action_GETREADY --> stateCar_READY
243  // action_START --> stateCar_RUNNING
244  // action_STOP --> stateCar_STARTUP
245  switch (juryActions(i8ActionID))
246  {
247  case action_GETREADY:
248  if(m_bDebugModeEnabled) LOG_INFO(cString::Format("State Controller: Received: Request Ready with maneuver ID %d",i16entry));
249  changeState(stateCar_READY);
250  setManeuverID(i16entry);
251  break;
252  case action_START:
253  if(m_bDebugModeEnabled) LOG_INFO(cString::Format("State Controller: Received: Run with maneuver ID %d",i16entry));
254  if (i16entry == m_i16CurrentManeuverID)
255  changeState(stateCar_RUNNING);
256  else
257  {
258  LOG_WARNING("StateController: The id of the action_START corresponds not with the id of the last action_GETREADY");
259  setManeuverID(i16entry);
260  changeState(stateCar_RUNNING);
261  }
262  break;
263  case action_STOP:
264  if(m_bDebugModeEnabled) LOG_INFO(cString::Format("State Controller: Received: Stop with maneuver ID %d",i16entry));
265  changeState(stateCar_STARTUP);
266  break;
267  }
268  }
269  }
270  RETURN_NOERROR;

392 tResult cStateController::changeState(stateCar newState)
393 {
394  // if state is the same do nothing
395  if (m_state == newState) RETURN_NOERROR;
396 
397  //to secure the state is sent at least one time
398  SendState(newState, m_i16CurrentManeuverID);
399 
400  //handle the timer depending on the state
401  switch (newState)
402  {
403  case stateCar_ERROR:
404  destroyTimer();
405  LOG_INFO("State ERROR reached");
406  break;
407  case stateCar_STARTUP:
408  destroyTimer();
409  LOG_INFO("State STARTUP reached");
410  break;
411  case stateCar_READY:
412  createTimer();
413  LOG_INFO(adtf_util::cString::Format("State READY reached (ID %d)",m_i16CurrentManeuverID));
414  break;
415  case stateCar_RUNNING:
416  if (m_state!=stateCar_READY)
417  LOG_WARNING("Invalid state change to Car_RUNNING. Car_READY was not reached before");
418  LOG_INFO("State RUNNING reached");
419  break;
420  case stateCar_COMPLETE:
421  destroyTimer();
422  LOG_INFO("State COMPLETE reached");
423  break;
424  }
425 
426  m_state = newState;
427  RETURN_NOERROR;
