114 tResult cQrReader::ProcessVideo(adtf::IMediaSample* pISample)
115 {
116  RETURN_IF_POINTER_NULL(pISample);
117 // creating new temporary variables
118  adtf_util::cImage oImage;
119  const tVoid* l_pSrcBuffer;
120 
121 // receiving data from inputsample, and saving to oImage
122  if (IS_OK(pISample->Lock(&l_pSrcBuffer)))
123  {
124  oImage.SetBits((tUInt8*) l_pSrcBuffer, &m_sInputFormat);
125  pISample->Unlock(l_pSrcBuffer);
126  }
127 
128 // convert adtf-image to greyscaled image which is used by zbar
129  RETURN_IF_FAILED(oImage.Convert(&m_sProcessFormat));
130 
131 // creates image which is used by the QR Scanner
132  Image image(m_sInputFormat.nWidth, m_sInputFormat.nHeight, "Y800", (unsigned char*)oImage.GetBitmap(), m_sInputFormat.nWidth * m_sInputFormat.nHeight);
133 
134 // scan image for codes and check for errors
135  int n = m_qrScanner.scan(image);
136  if (n== -1) LOG_ERROR(cString::Format("QR Reader: Error in QRScanner"));
137 
138 // extract results to console
139  for(Image::SymbolIterator symbol = image.symbol_begin();
140  symbol != image.symbol_end();
141  ++symbol) {
142  //print only new results
143  if (m_lastQRCode!=symbol->get_data())
144  {
145  if (m_bConsoleOutputEnabled) LOG_INFO(cString::Format("QR Reader: %s",symbol->get_data().c_str() ));
146  m_lastQRCode=symbol->get_data();
147  }
148  }
149 
150  // clean up
151  image.set_data(NULL, 0);
152 
153  RETURN_NOERROR;
154 }
