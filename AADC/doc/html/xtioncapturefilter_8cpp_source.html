158 tResult cXtionCaptureFilter::ReadData()
159 {
160  // check if state of thread and state of grabber is ok
161  if ((GetState() != State_Running) || (m_xtionGrabber.isRunning()!= tTrue))
162  RETURN_NOERROR;
163 
164  if (m_oDepthImage.IsConnected())
165  {
166  // create pointer to pixel image from xtion
167  openni::DepthPixel* ptrDepthImage;
168  tInt DepthImageSize=0;
169  // get pointer to data from xtion
170  if(m_xtionGrabber.getDepthFramePtr(ptrDepthImage,DepthImageSize))
171  {
172  //write date to opencv Mat
173  cv::Mat depth(m_sBitmapFormatDepth.nWidth,m_sBitmapFormatDepth.nHeight,CV_16UC1,ptrDepthImage);
174  // multiply the 11-bit depth values by 32 to extend the color range to a full 16-bits altough 16 bit is set in grabber
175  depth.convertTo(depth, -1, 32);
176  // transmit data in media sample over the output pin
177  cObjectPtr<IMediaSample> pNewDepthSample;
178  if (IS_OK(AllocMediaSample(&pNewDepthSample)))
179  {
180  tTimeStamp tmStreamTime = _clock ? _clock->GetStreamTime() : adtf_util::cHighResTimer::GetTime();
181  pNewDepthSample->Update(tmStreamTime, depth.data, tInt32(DepthImageSize), 0);
182  m_oDepthImage.Transmit(pNewDepthSample);
183  }
184  }
185  }
186  if (m_oRGBVideo.IsConnected())
187  {
188  // create pointer to pixel image from xtion
189  openni::RGB888Pixel* ptrColorImage;
190  tInt colorImageSize=0;
191  // get pointer to data from xtion
192  if (m_xtionGrabber.getColorFramePtr(ptrColorImage,colorImageSize))
193  {
194  //write date to opencv Mat
195  cv::Mat colorImage(m_sBitmapFormat.nWidth,m_sBitmapFormat.nHeight,CV_8UC3,ptrColorImage);
196  //convert from BGR to RGB
197  cv::cvtColor(colorImage, colorImage, CV_BGR2RGB);
198  // transmit data in media sample over the output pin
199  cObjectPtr<IMediaSample> pNewRGBSample;
200  if (IS_OK(AllocMediaSample(&pNewRGBSample)))
201  {
202  tTimeStamp tmStreamTime = _clock ? _clock->GetStreamTime() : adtf_util::cHighResTimer::GetTime();
203  pNewRGBSample->Update(tmStreamTime, colorImage.data, tInt32(colorImageSize), 0);
204  m_oRGBVideo.Transmit(pNewRGBSample);
205  }
206  }
207  }
208  RETURN_NOERROR;
209 }
