185 tFloat32 RPMCalculation::calcLatestRpm( tFloat32 counterValue, tFloat32 lastCounterValue,tTimeStamp currentTimestamp, tTimeStamp lastTimestamp )
186 {
187  long deltaTime = long(currentTimestamp - lastTimestamp); //microseconds
188  unsigned long deltaCounter = long(counterValue) - long(lastCounterValue);
189  tFloat32 deltaRounds = tFloat32(deltaCounter)/8.0f; //one round = 8 counts
190  tFloat32 rpm;
