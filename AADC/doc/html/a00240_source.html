189 tResult cRoadsignDetectionFilter::ProcessVideo(adtf::IMediaSample* pISample)
190 {
191 
192  RETURN_IF_POINTER_NULL(pISample);
193 
194  //creating new media sample for output
195  cObjectPtr<IMediaSample> pNewSample;
196  RETURN_IF_FAILED(_runtime->CreateInstance(OID_ADTF_MEDIA_SAMPLE, IID_ADTF_MEDIA_SAMPLE, (tVoid**) &pNewSample));
197  RETURN_IF_FAILED(pNewSample->AllocBuffer(m_sOutputFormat.nSize));
198 
199  //create matrixes for in and output
200  Mat inputFrame, inputFrameIHLS;
201  Mat outputFrame, outputFrameRGB, outputFrameRed, outputFrameBlue;
202  //creating new pointer for input data
203  const tVoid* l_pSrcBuffer;
204 
205  // set to false to check if any sign was detected in this frame, possibly set to true in sendRoadSignStruct
206  m_SignInFrame = tFalse;
207 
208  //receiving data from inputsample, and saving to inputFrame
209  if (IS_OK(pISample->Lock(&l_pSrcBuffer)))
210  {
211  //convert to mat
212  inputFrame = Mat(m_sInputFormat.nHeight,m_sInputFormat.nWidth,CV_8UC3,(tVoid*)l_pSrcBuffer,m_sInputFormat.nBytesPerLine);
213  if (m_outputMode==2) outputFrameRGB = inputFrame.clone();
214  //convert to ihls color space
215  inputFrameIHLS = convert_rgb_to_ihls(inputFrame);
216 
217  //vectors for contours and their hierarchy
218  vector<vector<Point> > contours;
219  vector<Vec4i> hierarchy;
220 
221  // convert to nhs colorspace with a red filter
222  outputFrameRed = convert_ihls_to_nhs_red(inputFrameIHLS,m_red_hue_max,m_red_hue_min,m_red_sat_min); //convert image to nhs color space (for red color)
223 
225  findContours( outputFrameRed, contours, RETR_EXTERNAL, CV_CHAIN_APPROX_SIMPLE, Point(0, 0) );
226  for( tUInt i = 0; i< contours.size(); i++ )
227  {
228  //check if countour area is in range of minimum and maximum area
229  if (fabs(contourArea(contours[i]))>m_minArea && fabs(contourArea(contours[i]))<m_maxArea)
230  {
231  //check if width and height of area has proper ratio
232  Rect boundRect = boundingRect( Mat(contours[i]) );
233  tFloat32 contourAspectRatio = ((tFloat32)boundRect.width/(tFloat32)boundRect.height); //calculates aspect ratio between height and width
234 
235  if ((contourAspectRatio<1+m_errorFactorRatio) && (contourAspectRatio>1-m_errorFactorRatio) )
236  {
237  vector<Point> contours_poly;
238  approxPolyDP(contours[i], contours_poly, m_smoothFactorCorner, tTrue);
239  //check for triangle
240  if(contours_poly.size() == 3)
241  {
242  //TODO unterscheidung der dreieckigen Zeichen
243  sendRoadSignStruct(1,fabs(contourArea(contours[i])/m_maxArea),pISample->GetTime());
244  if (m_outputMode==2) drawContours(outputFrameRGB, contours, i, Scalar(0, 0, 255),2, 8, hierarchy, 0, Point() );
245  }
246  //check for octangle
247  else if (contours_poly.size() == 8)
248  {
249  if (m_outputMode==2) drawContours(outputFrameRGB, contours, i, Scalar(0, 0, 255),2, 8, hierarchy, 0, Point() );
250  sendRoadSignStruct(3,fabs(contourArea(contours[i])/m_maxArea),pISample->GetTime());
251  }
252  }
253  }
254  }
255 
256  contours.clear();
257  hierarchy.clear();
258 
259  // convert to nhs colorspace with a blue filter
260  outputFrameBlue = convert_ihls_to_nhs_blue(inputFrameIHLS,m_blue_hue_max,m_blue_hue_min,m_blue_sat_min);
261 
263  findContours( outputFrameBlue, contours, RETR_EXTERNAL, CV_CHAIN_APPROX_SIMPLE, Point(0, 0) );
264  for( tUInt i = 0; i< contours.size(); i++ )
265  {
266  //check if countour area is in range of minimum and maximum area
267  if (fabs(contourArea(contours[i]))>m_minArea && fabs(contourArea(contours[i]))<m_maxArea)
268  {
269  //check if width and height of area has proper ratio
270  Rect boundRect = boundingRect( Mat(contours[i]) );
271  tFloat32 contourAspectRatio = ((tFloat32)boundRect.width/(tFloat32)boundRect.height); //calculates aspect ratio between height and width
272 
273  if ((contourAspectRatio<1+m_errorFactorRatio) && (contourAspectRatio>1-m_errorFactorRatio) )
274  {
275  vector<Point> contours_poly;
276  approxPolyDP(contours[i], contours_poly, m_smoothFactorCorner, tTrue);
277  //check for rectangle
278  if (contours_poly.size() == 4)
279  {
280  tFloat64 estimated_area = boundRect.width * boundRect.height;
281  if (fabs(fabs(contourArea(contours[i])) - estimated_area)<(estimated_area*m_errorFactorRect))
282  {
283  if (m_outputMode==2) drawContours(outputFrameRGB, contours, i, Scalar(255, 0, 0),2, 8, hierarchy, 0, Point() );
284  sendRoadSignStruct(4,fabs(contourArea(contours[i])/m_maxArea),pISample->GetTime());
285  }
286  }
287  //check for circle
288  else
289  {
290  tFloat64 estimated_area = M_PI * boundRect.width / 2 * boundRect.height / 2;
291  if (fabs(fabs(contourArea(contours[i])) - estimated_area)<(estimated_area*m_errorFactorCircle))
292  {
293  if (m_outputMode==2) drawContours(outputFrameRGB, contours, i, Scalar(255, 0, 0),2, 8, hierarchy, 0, Point() );
294  sendRoadSignStruct(5,fabs(contourArea(contours[i])/m_maxArea),pISample->GetTime());
295  }
296  }
297  }
298  }
299  }
300  //set output frame depending on property
301  if (m_outputMode==2) outputFrame = outputFrameRGB;
302  else if (m_outputMode==3) outputFrame = outputFrameRed;
303  else if (m_outputMode==4) outputFrame = outputFrameBlue;
304  //setting parameters of output frame
305  if ((outputFrame.elemSize() * outputFrame.total())!=(unsigned int)m_sOutputFormat.nSize && m_outputMode!=1)
306  {
307  tBitmapFormat newOutputFormat = m_sOutputFormat ;
308  newOutputFormat.nSize = (outputFrame.elemSize() * outputFrame.total());
309  newOutputFormat.nWidth = outputFrame.cols;
310  newOutputFormat.nHeight = outputFrame.rows;
311  newOutputFormat.nBitsPerPixel = outputFrame.elemSize()*8;
312  newOutputFormat.nBytesPerLine = outputFrame.step;
313  newOutputFormat.nPixelFormat = adtf_util::cImage::PF_8BIT;
314  newOutputFormat.nPaletteSize = 0;
315  UpdateOutputImageFormat(&newOutputFormat);
316 
317  }
318  }
319 
320  pISample->Unlock(l_pSrcBuffer);
321 
322  //update new media sample with image data if something has ti be transmitted
323  if (m_outputMode!=1)
324  {
325  pNewSample->Update(pISample->GetTime(), outputFrame.data, m_sOutputFormat.nSize, 0);
326  m_oPinOutputVideo.Transmit(pNewSample);
327  }
328  // send a sample which indicates that no sign was detectec in this frame
329  if (!(m_SignInFrame)) sendRoadSignStruct(0,0,pISample->GetTime());
330 
331  RETURN_NOERROR;
332 }

379

383  RETURN_IF_FAILED(m_oPinRoadSign.Transmit(pMediaSample));
384 
385  if (m_bDebugModeEnabled) LOG_INFO(cString::Format("Zeichen ID %d erkannt. Area: %f",ID,ImageSize));
386 
387  RETURN_NOERROR;
