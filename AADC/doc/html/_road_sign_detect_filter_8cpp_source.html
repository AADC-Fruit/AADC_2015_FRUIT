157 tResult cRoadsignDetectionFilter::ProcessVideo(adtf::IMediaSample* pISample)
158 {
159 
160  RETURN_IF_POINTER_NULL(pISample);
161 //creating new temporary variables
162  adtf_util::cImage oImage;
163  const tVoid* l_pSrcBuffer;
164 
165 //receiving data from inputsample, and saving to oImage
166  if (IS_OK(pISample->Lock(&l_pSrcBuffer)))
167  {
168  oImage.SetBits((tUInt8*) l_pSrcBuffer, &m_sInputFormat);
169  pISample->Unlock(l_pSrcBuffer);
170  }
171 
172 //sending data to dll and processing data in DLL
173  tInt Result = 0;
174  Result = interfaceClass->update((tUInt8*)oImage.GetBitmap());
175 
176 
177  if (Result !=0)
178  {
179  cString newSign = convertIntToEnum(Result);
180  if (m_bDebugModeEnabled) LOG_INFO(adtf_util::cString::Format("Roadsign Detection Filter: Erkanntes Verkehrszeichen: %s",newSign.GetPtr()));
181 
182  if (m_RoadSignResult != Result)
183  {
184  m_RoadSignResult = Result;
185 
186  //transmit number of detected sign
187  cObjectPtr<IMediaSample> pNewSample;
188  RETURN_IF_FAILED(AllocMediaSample((tVoid**)&pNewSample));
189  RETURN_IF_FAILED(pNewSample->Update(cHighResTimer::GetTime(), &m_RoadSignResult, sizeof(tInt), IMediaSample::MSF_None));
190  RETURN_IF_FAILED(m_oPinSignNumber.Transmit(pNewSample));
191  }
192  }
193 
194 //checking if output image of dll has changed in size
195 
196  if (m_sOutputFormat.nSize != (tInt)interfaceClass->getProcImgSize())
197  {
198  m_sOutputFormat.nBytesPerLine = (tInt)interfaceClass->getProcImgBytesPerLine();
199  m_sOutputFormat.nSize =(tInt)interfaceClass->getProcImgSize();
200  m_sOutputFormat.nBitsPerPixel=(tInt)interfaceClass->getProcImgBitsPerPixel();
201  m_sOutputFormat.nPaletteSize = 0;
202  m_sOutputFormat.nPixelFormat = adtf_util::cImage::PF_8BIT ;
203  m_oPinOutputVideo.SetFormat(&m_sOutputFormat, NULL);
204  }
205 
206 //creating new media sample with processed data from dll and sending frame via pin
207  cObjectPtr<IMediaSample> pNewSample;
208  RETURN_IF_FAILED(AllocMediaSample((tVoid**)&pNewSample));
209  RETURN_IF_FAILED(pNewSample->AllocBuffer(m_sOutputFormat.nSize));
210  //setting timestamp of media sample
211  tTimeStamp tmTime = (_clock != NULL ? _clock->GetStreamTime() : cHighResTimer::GetTime());
212 
213  RETURN_IF_FAILED(pNewSample->SetTime(tmTime));
214 
215  if (IS_OK(pNewSample->CopyBufferFrom((tVoid*)interfaceClass->getProcessedData(),m_sOutputFormat.nSize)))
216  {
217  RETURN_IF_FAILED(m_oPinOutputVideo.Transmit(pNewSample));
218  }
219 
220  RETURN_NOERROR;
221 }
