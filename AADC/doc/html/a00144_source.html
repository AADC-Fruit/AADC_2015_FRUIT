406 tResult cArduinoSensors::Process_ArduinoCOMInputPin(IMediaSample *pMediaSample)
407 {
408  if (pMediaSample != NULL && m_pCoderArduinoData != NULL)
409  {
410  //write values with zero
411  tUInt8 SOF = 0;
412  tUInt8 ID = 0;
413  tUInt32 arduinoTimestamp = 0;
414  tUInt8 DataLength = 0;
415  tUInt8 frameData[25];
416 
417  cMemoryBlock::MemSet(frameData, 0x00, sizeof(frameData));
418 
419  { // focus for sample read lock
420  // read-out the incoming Media Sample
421  __adtf_sample_read_lock_mediadescription(m_pCoderArduinoData,pMediaSample,pCoder);
422 
423  //get values from media sample
424  pCoder->Get("ui8SOF", (tVoid*)&SOF);
425  pCoder->Get("ui8ID", (tVoid*)&ID);
426  pCoder->Get("ui32ArduinoTimestamp", (tVoid*)&arduinoTimestamp);
427  pCoder->Get("ui8DataLength", (tVoid*)&DataLength);
428  pCoder->Get("ui8Data", (tVoid*)&(frameData));
429  }
430 
431  if (m_bDebugModeEnabled) LOG_INFO(cString::Format("Sensorfilter received: ID %x DataLength %d",ID,DataLength));
432 
433 
434  switch(ID)
435  {
436  case (ID_ARD_SENS_WHEELENC):
437  RETURN_IF_FAILED(TransmitCountRPMData(pMediaSample->GetTime(),arduinoTimestamp,frameData));
438  break;
439  case (ID_ARD_SENS_US):
440  RETURN_IF_FAILED(TransmitUsData(pMediaSample->GetTime(),arduinoTimestamp,frameData));
441  break;
442  case (ID_ARD_SENS_IMU):
443  RETURN_IF_FAILED(TransmitAccData(pMediaSample->GetTime(),arduinoTimestamp,frameData));
444  RETURN_IF_FAILED(TransmitGyroData(pMediaSample->GetTime(),arduinoTimestamp,frameData));
445  break;
446  case (ID_ARD_SENS_VOLTAGE):
447  RETURN_IF_FAILED(TransmitVoltData(pMediaSample->GetTime(),arduinoTimestamp,frameData));
448  break;
449  case (ID_ARD_SENS_IR):
450  RETURN_IF_FAILED(TransmitIrData(pMediaSample->GetTime(),arduinoTimestamp,frameData));
451  break;
452  case (ID_ARD_SENS_STEER_ANGLE):
453  RETURN_IF_FAILED(TransmitSteeringAngleData(pMediaSample->GetTime(),arduinoTimestamp,frameData));
454  break;
455  default:
456  break;
457  }
458  }
459 
460  RETURN_NOERROR;
461 }
