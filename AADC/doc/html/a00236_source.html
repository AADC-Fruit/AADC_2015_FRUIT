83 retrieve_theta(tUInt r, tUInt g, tUInt b)
84 {
85  tFloat32 theta;
86 
87  // The numerator part of equation
88  tFloat32 numerator = r - (g * 0.5f) - (b * 0.5f);
89 
90  // The denominator part of equation
91  tFloat32 denominator = static_cast<tFloat32> ((r * r) + (g * g) + (b * b) - (r * g) - (r * b) - (g * b));
92 
93  tFloat32 temp = numerator / sqrtf(denominator);
94  theta = acos(temp);
95 
96  return theta;
97 }
