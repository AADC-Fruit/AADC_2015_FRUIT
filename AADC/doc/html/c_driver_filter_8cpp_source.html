122 {
123  RETURN_IF_POINTER_NULL(pMediaSample);
124  RETURN_IF_POINTER_NULL(pSource);
125 
126  if (nEventCode == IPinEventSink::PE_MediaSampleReceived )
127  {
128  if (pSource == &m_RunInputPin && m_pCoderDescRun != NULL)
129  {
130 
131  cObjectPtr<IMediaCoder> pCoder;
132  RETURN_IF_FAILED(m_pCoderDescRun->Lock(pMediaSample, &pCoder));
133 
134  tInt16 entry = -1;
135 
136  pCoder->Get("i16ManeuverEntry", (tVoid*)&entry);
137  m_pCoderDescRun->Unlock(pCoder);
138  emit sendRun(entry);
139  if(m_bDebugModeEnabled) LOG_INFO(cString::Format("Driver Module: Received Run with Maneuver ID %d",entry));
140  }
141  else if (pSource == &m_StopInputPin && m_pCoderDescStopIn != NULL)
142  {
143 
144  cObjectPtr<IMediaCoder> pCoder;
145  RETURN_IF_FAILED(m_pCoderDescStopIn->Lock(pMediaSample, &pCoder));
146 
147  tBool value = false;
148 
149  pCoder->Get("bStopFlag", (tVoid*)&value);
150  m_pCoderDescStopIn->Unlock(pCoder);
151  emit sendStop();
152  if(m_bDebugModeEnabled) LOG_INFO(cString::Format("Driver Module: Receiver Stop flag: %d",value));
153  }
154 
155  }
156  RETURN_NOERROR;
157
