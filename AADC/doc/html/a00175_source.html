178 tResult cDriverFilter::OnSendState(stateCar stateID, tInt16 i16ManeuverEntry)
179 {
180  cObjectPtr<IMediaSample> pMediaSample;
181  RETURN_IF_FAILED(AllocMediaSample((tVoid**)&pMediaSample));
182 
183  cObjectPtr<IMediaSerializer> pSerializer;
184  m_pCoderDescDriverStruct->GetMediaSampleSerializer(&pSerializer);
185  tInt nSize = pSerializer->GetDeserializedSize();
186 
187  tInt8 value = tInt8(stateID);
188 
189  RETURN_IF_FAILED(pMediaSample->AllocBuffer(nSize));
190  { // focus for sample write lock
191  __adtf_sample_write_lock_mediadescription(m_pCoderDescDriverStruct,pMediaSample,pCoder);
192 
193 
194  pCoder->Set("i8StateID", (tVoid*)&value);
195  pCoder->Set("i16ManeuverEntry", (tVoid*)&i16ManeuverEntry);
196  }
197 
198  pMediaSample->SetTime(_clock->GetStreamTime());
199  m_DriverStructOutputPin.Transmit(pMediaSample);
200  if(m_bDebugModeEnabled)
201  {
202  switch (stateID)
203  {
204  case stateCar_READY:
205  LOG_INFO(cString::Format("Driver Module: Send state: READY, Maneuver ID %d",i16ManeuverEntry));
206  break;
207  case stateCar_RUNNING:
208  LOG_INFO(cString::Format("Driver Module: Send state: RUNNING, Maneuver ID %d",i16ManeuverEntry));
209  break;
210  case stateCar_COMPLETE:
211  LOG_INFO(cString::Format("Driver Module: Send state: COMPLETE, Maneuver ID %d",i16ManeuverEntry));
212  break;
213  case stateCar_ERROR:
214  LOG_INFO(cString::Format("Driver Module: Send state: ERROR, Maneuver ID %d",i16ManeuverEntry));
215  break;

217  break;
218  }

252  {

261  {

265  sector.maneuverList.push_back(man);

275  }
