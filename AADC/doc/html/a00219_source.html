156 tResult cJuryTransmitter::OnPinEvent(IPin* pSource, tInt nEventCode, tInt nParam1, tInt nParam2, IMediaSample* pMediaSample)
157 {
158  RETURN_IF_POINTER_NULL(pMediaSample);
159  RETURN_IF_POINTER_NULL(pSource);
160 
161  if (nEventCode == IPinEventSink::PE_MediaSampleReceived && m_pCoderDescNotAus != NULL && pSource == &m_DriverStructInputPin)
162  {
163  // read-out the incoming Media Sample
164  cObjectPtr<IMediaCoder> pCoder;
165  RETURN_IF_FAILED(m_pCoderDescDriverStruct->Lock(pMediaSample, &pCoder));
166 
167  tInt8 stateDriver = false;
168  tInt16 entry = -1;
169 
170  pCoder->Get("i8StateID", (tVoid*)&stateDriver);
171  pCoder->Get("i16ManeuverEntry", (tVoid*)&entry);
172  m_pCoderDescDriverStruct->Unlock(pCoder);
173  //update the gui
174  emit sendDriverState(stateDriver, entry);
175  //update the timeline
176  if ( m_last_timestamp !=0 && (m_last_entryId < entry || m_last_entryId == -1 || stateDriver == -1 || stateDriver == 2))
177  {
178  tTimeStamp timeDiff = (pMediaSample->GetTime() - m_last_timestamp)/1000; //time in milliseconds
179  m_last_timestamp = pMediaSample->GetTime();
180  QString Message;
181  if (stateDriver==0)
182  Message = "from " + QString::number(m_last_entryId) + " to " + QString::number(entry) + ": " + QString::number(timeDiff) + " msec: Ready";
183  else if (stateDriver==1)
184  Message = "from " + QString::number(m_last_entryId) + " to " + QString::number(entry) + ": " + QString::number(timeDiff) + " msec: OK";
185  else if (stateDriver==2)
186  Message = "from " + QString::number(m_last_entryId) + " to " + QString::number(entry) + ": " + QString::number(timeDiff) + " msec: Complete";
187  else
188  Message = "from " + QString::number(m_last_entryId) + " to " + QString::number(entry) + ": " + QString::number(timeDiff) + " msec: Error";
189  m_last_entryId = entry;
190  sendMessage(Message);
191  }
192  else
193  m_last_timestamp = pMediaSample->GetTime();
194 
195  if(m_bDebugModeEnabled) LOG_INFO(cString::Format("Jury Module: State value: %d with index %d",stateDriver,entry));
196 
197  }
198  RETURN_NOERROR;
199 
200 }
